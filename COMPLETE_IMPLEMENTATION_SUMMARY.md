import { usePainTrackerStore } from '@/stores/pain-tracker-store';

function SmartPainDashboard() {
  // Retention features
  const getDailyPrompt = usePainTrackerStore(s => s.getDailyPrompt);
  const getPendingInsights = usePainTrackerStore(s => s.getPendingInsights);
  const getWinConditions = usePainTrackerStore(s => s.getWinConditions);
  
  // Intelligence features
  const getPredictiveInsights = usePainTrackerStore(s => s.getPredictiveInsights);
  const getMultiVariateAnalysis = usePainTrackerStore(s => s.getMultiVariateAnalysis);
  const getEnhancedPatterns = usePainTrackerStore(s => s.getEnhancedPatterns);
  const getSmartRecommendations = usePainTrackerStore(s => s.getSmartRecommendations);
  
  // Get all insights
  const prompt = getDailyPrompt();
  const pendingInsights = getPendingInsights();
  const winConditions = getWinConditions();
  const predictions = getPredictiveInsights();
  const analysis = getMultiVariateAnalysis();
  const patterns = getEnhancedPatterns();
  const recommendations = getSmartRecommendations();
  
  return (
    <div className="smart-dashboard">
      {/* Daily engagement */}
      <DailyCheckInPrompt prompt={prompt} />
      <ReturnIncentiveWidget pendingInsights={pendingInsights} />
      
      {/* Predictions */}
      <PredictionCard prediction={predictions.nextDayPrediction} />
      <TrendForecast trend={predictions.next7DaysTrend} />
      
      {/* Patterns */}
      <PatternsSection patterns={patterns} />
      <CorrelationsChart analysis={analysis} />
      
      {/* Actions */}
      <RecommendationsPanel recommendations={recommendations} />
      <ActionPlanTracker plans={recommendations.actionPlans} />
      
      {/* Progress */}
      <WinConditionsDisplay conditions={winConditions} />
      <IdentityDashboard />
    </div>
  );
}

### Quick Prediction Widget

```typescript
function TomorrowWidget() {
  const insights = usePainTrackerStore(s => s.getPredictiveInsights());
  
  if (!insights.nextDayPrediction) {
    return <EmptyState message="Track 7+ days to unlock predictions" />;
  }
  
  const { predictedLevel, confidence, range, factors } = insights.nextDayPrediction;
  
  return (
    <Card className="tomorrow-widget">
      <h3>Tomorrow's Forecast</h3>
      <div className="prediction">
        <span className="level">{predictedLevel.toFixed(1)}</span>
        <span className="scale">/10</span>
      </div>
      <div className="confidence">
        {(confidence * 100).toFixed(0)}% confident
      </div>
      <div className="range">
        Range: {range.min.toFixed(1)} - {range.max.toFixed(1)}
      </div>
      <ul className="factors">
        {factors.map(factor => <li key={factor}>{factor}</li>)}
      </ul>
    </Card>
  );
}
```
